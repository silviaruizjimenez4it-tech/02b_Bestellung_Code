    anzahlCheeseburger = 0
    anzahlChiliBurger = 0
    anzahlPommes = 0

    WHILE Bestellung nicht abgeschlossen

        PRINT "Einzelartikel (1) oder Menü (2) oder Beenden (0)?"
        eingabe = NutzerInput

        IF eingabe == 0
            BREAK  // Bestellung beenden

        ELSE IF eingabe == 1
            // Einzelartikel auswählen
            PRINT "Was möchten Sie bestellen? Hamburger(1), Cheeseburger(2), ChiliBurger(3), Pommes(4), Cola(5)"
            artikel = NutzerInput

            PRINT "Wie viele möchten Sie?"
            menge = NutzerInput

                CASE 2: anzahlCheeseburger += menge
